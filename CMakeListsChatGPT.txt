# Set the path to the SFML library directory

set(SFML_INCLUDE_DIR ${CMAKE_CURRENT_SOURCE_DIR}/include/${THE_OS}-SFML-2.5.1/include)

# Include SFML headers

)

    IMPORTED_LOCATION ${THE_SFML_LIBRARY_DIR}/sfml-window.a

)

    IMPORTED_LOCATION ${THE_SFML_LIBRARY_DIR}/sfml-system.a
